```scala
package eu.jpmena

object Account {
  private var lastNumber = 0
  def newUniqueNumber() = {
    lastNumber += 1;
    lastNumber
  }
}


class Account {
  val id = Account.newUniqueNumber()
  private var balance = 0.0

  def deposit(amount: Double) {
    balance += amount
  }

  def withDram(amount: Double) {
    if (balance - amount < 0) {
    } else {
      balance -= amount
    }
  }

  def representation = s"for the account $id the balance is $balance"

}
```
