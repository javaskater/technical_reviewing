## playing with singletons in the REPL

* no _new_ keyword because Accounts is a singleton object, (the constructor is called only the first time this object is used, in our cse with o1)
* the _assert_ confirms that o1 and o2 are the same instance
```scala
scala> object Accounts {
     | private var lastNumber = 0
     | def newUniqueNumber() = {lastNumber += 1; lastNumber}
     | }
defined object Accounts

scala> val o1 = Accounts
o1: Accounts.type = Accounts$@2e04c51a

scala> o1.newUniqueNumber
res50: Int = 1

scala> o1.newUniqueNumber
res51: Int = 2

scala> val o2 = Accounts
o2: Accounts.type = Accounts$@2e04c51a

scala> assert (o1 == o2)

scala> o2.newUniqueNumber
res53: Int = 3
//I can call directly the Singleton fabric
scala> Accounts.newUniqueNumber
res54: Int = 4

scala> o2.newUniqueNumber
res55: Int = 5
```
