```scala
package com {
  package horstmann {
    object Utils {
      def percentOf(value: Double, rate: Double): Double = value * rate / 100
    }


    package impatient { //child of the horstmann package

      import scala.collection.mutable

      class Employee {
        var salary: scala.Double = 0.0

        def giveRaise(rate: scala.Double): Unit = {
          salary += Utils.percentOf(salary, rate)
        }
      }

      class Manager {
        import scala.collection.mutable._
        val subordinates = new ArrayBuffer[Employee]
      }
    }

  }
}
```
> By putting the imports where they are needed, you can greatly reduce the potential for conflicts.
