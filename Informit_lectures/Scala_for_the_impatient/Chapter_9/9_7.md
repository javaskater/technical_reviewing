* IntelliJ says taht _val entries: Stream[Path]_
* That chapter shows also shows to handle a _java.util.Stream_ interface
  * with a **forEach**
  * with a *try finally* 
```scala
package com.absile.scala.chap09
import java.nio.file._

object Test9_7 extends App{
  val dirname = "/home/jmena01/ERICA/TOPAD/TOPAD-Cible"
  val entries = Files.walk(Paths.get(dirname)) //recursively
  //val entries = Files.list(Paths.get(dirname)) //only the dir not the sub-dirs
  try {
    entries.forEach(p => println(p.toAbsolutePath.toString))
  } finally {
    entries.close()
  }
}
```
