* Obliged to use Java classes, Scala has nothing special for reading binary files
```scala
package com.absile.scala.chap09

import java.io.{File, FileInputStream}

object Test9_5 extends App {
  val file = new File("/home/jmena01/ERICA/TOPAD/TOPAD-Cible/recettesRFFevrier.csv")
  val in = new FileInputStream(file)
  val file_size_in_bytes = file.length.toInt
  val bytes = new Array[Byte](file_size_in_bytes)
  val numberOfBytesRead = in.read(bytes)
  println(s"I read ${numberOfBytesRead} bytes")
  assert(numberOfBytesRead == file_size_in_bytes)
}
```
