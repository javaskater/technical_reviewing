import scala.io.Source

object Test9_2 extends App{
  val source = Source.fromFile("/home/jmena01/ERICA/TOPAD/TOPAD-Cible/recettesRFFevrier.csv", "UTF-8")

  var index_a:Int = 0
  while (iter.hasNext){ //you read a character without consuming it
    if (iter.head == 'a'){
       index_a += 1
       println(s"c'est le ${index_a} eme character ${iter.next}")
    } else {
      iter.next() //you read and consume the character at the same time
    }
  }
}
