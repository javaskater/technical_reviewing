# _.r_ is still an [implicit conversion](https://docs.scala-lang.org/tour/implicit-conversions.html)
* From String to RegExp!
## Simple use of RegExp
```scala
package com.absile.scala.chap09

opackage com.absile.scala.chap09

object test9_10 extends App {
  val numPattern = "[0-9]+".r
  for (matchString <- numPattern.findAllIn("99 bottles, 98 bottles")){
    println(s"[main][loop] found |$matchString|")
  }
  val matches = numPattern.findAllIn("99 bottles, 98 bottles").toArray
  println(s"[main]The Array has size ${matches.size}")
  val wsnumwsPattern = """\s+[0-9]+\s+""".r
  for (matchString <- wsnumwsPattern.findAllIn("99 bottles, 98 bottles Hello")){
    println(s"[main][loop2] found |$matchString|")
  }
  val firstMatch = wsnumwsPattern.findFirstMatchIn("99 bottles, 98 bottles Hello")
  println(f"[main]First match ${firstMatch.getOrElse("Inconnu")}")

  val pairMatches = numPattern.replaceSomeIn("99 bottles, 98 bottles", m => {if (m.matched.toInt % 2 == 0) Some("XX") else None})
  println(f"[main]pair matches |${pairMatches}|")
}
```

* and the lift function
```scala
package com.absile.scala.chap09

object Test9_10_1 extends App{
  //We are obliged to put the String* at the end of the parameter list
  def renvoyer(i:Int, args:String*):Option[String]={
    args.lift(i)
  }
  println(s"[main] ${renvoyer(2, "Bonjour", "jpm", "mena")}")
  println(s"[main] ${renvoyer(3, "Bonjour", "jpm", "mena").getOrElse("Unknown")}")
}
```
